import java.sql.*;
import java.util.Scanner;

public class menu {
    public static void main(String[] args) throws Exception {
        Connection c = DriverManager.getConnection("jdbc:mysql://localhost:3306/test", "root", "123456");
        Statement s = c.createStatement();
        Scanner sc = new Scanner(System.in);

        // Create table if not exists
        s.executeUpdate("CREATE TABLE IF NOT EXISTS employee (id INT, name VARCHAR(30))");

        while (true) {
            System.out.println("\n--- MENU ---");
            System.out.println("1) Add User");
            System.out.println("2) Update User");
            System.out.println("3) Delete User");
            System.out.println("4) Exit");
            System.out.print("Enter choice: ");
            int choice = sc.nextInt();
            sc.nextLine(); // consume newline

            switch (choice) {
                case 1:
                    System.out.print("Enter ID: ");
                    int id = sc.nextInt();
                    sc.nextLine();
                    System.out.print("Enter Name: ");
                    String name = sc.nextLine();
                    s.executeUpdate("INSERT INTO employee VALUES (" + id + ", '" + name + "')");
                    System.out.println("User added.");
                    break;

                case 2:
                    System.out.print("Enter ID to update: ");
                    int updateId = sc.nextInt();
                    sc.nextLine();
                    System.out.print("Enter new name: ");
                    String newName = sc.nextLine();
                    s.executeUpdate("UPDATE employee SET name = '" + newName + "' WHERE id = " + updateId);
                    System.out.println("User updated.");
                    break;

                case 3:
                    System.out.print("Enter ID to delete: ");
                    int deleteId = sc.nextInt();
                    s.executeUpdate("DELETE FROM employee WHERE id = " + deleteId);
                    System.out.println("User deleted.");
                    break;

                case 4:
                    System.out.println("Exiting...");
                    c.close();
                    sc.close();
                    return;

                default:
                    System.out.println("Invalid choice. Try again.");
            }

            // Show current table
            ResultSet rs = s.executeQuery("SELECT * FROM employee");
            System.out.println("\nCurrent Records:");
            while (rs.next()) {
                System.out.println(rs.getInt(1) + " " + rs.getString(2));
            }
        }
    }
}
